\begin{enumerate}

\end{enumerate}

   \section{Аналитическая часть} 
\end{center}

Расстояние между строками a и b можно найти при помощи некоторых редакционных расстояний. Рассмотрим их.

Расстояние Левенштейна --- это метрика cходства между двумя строковыми последовательностями \cite{mpl}. Оно может быть рассчитано по формуле ~\ref{eq:Lev}:\\
\begin{equation} \label{eq:Lev}
d_{a, b}(i,j) =
\begin{cases}
    0 & i = 0, j = 0
\\   i & j = 0, i > 0
\\   j & i = 0, j > 0
\\ \min \begin{cases}
    d_{a, b}(i, j - 1) + 1
\\  d_{a, b}(i - 1, j) + 1
\\  d_{a, b}(i - 1, j - 1) + m(a[i], b[j])
\end{cases}  & j > 0, i > 0  
\end{cases}\\
\end{equation}

Нетрудно понять, что первые три строчки формулы заполнят первый столбец и первую строку матрицы. Для всех остальных ячеек матрицы используется четветрая формула. Стоит уделить внимание оператору ``m(a[i], b[j])''. Он сравнивает символы в слове, и, если они одинаковые, он дает 0, иначе --- 1.


Расстояние Дамерау-Левенштейна вычисляется по формуле ~\ref{eq:Levv}:
\begin{equation} \label{eq:Levv}
d_{a, b}(i, j) = 
\begin{cases} 
\max(i, j) & \mbox{если } i \cdot j = 0;
\\ \min \begin{cases} 
d_{a, b}(i - 1, j) + 1
\\d_{a, b}(i, j - 1) + 1
\\d_{a, b}(i - 1, j - 1 ) + 1_{(a_i = b_j)}
\\d_{a, b}(i - 2, j - 2) + 1
\end{cases} & \mbox{если } i,j > 1, a_i = b_{j - 1}, a_{i - 1} = b_j;
\\ \min \begin{cases} 
d_{a, b}(i - 1, j) + 1
\\d_{a, b}(i, j - 1) + 1
\\d_{a, b}(i - 1, j - 1) + 1_{(a_i = b_j)}
\end{cases} & \mbox{иначе} 
\end{cases}\\
\end{equation}

\justifying
Значительным отличием в нахождении расстояния Дамерау-Левенштейна от расстояния Левенштейна является транспозиция соседних символов. 

Говоря об эффективности методов вычисления расстояния слов, хочется отметить рекурсивный способ Дамерау-Левенштейна с кэшем. Поскольку после каждого подсчета порождается еще три вычисления, рекурсия с кэшем может помочь сохранить процессорное время. Поэтому вместо $3^{k*l}$ раз, где k и l длина слов соответсвенно, нужно вычислить $k * l$ раз.
Что касается других способов решения задачи, можно использовать нерекурсивный метод с заполнением матрицы размера $(k+1)*(l+1)$. Однако можно заметить, что для вычисления расстояния Левенштейна используются только числа предыдущей и вычисляемой строки.  Поэтому в каждый момент вычисления можно хранить две строки матрицы.

int levenstein(string str_1, string str_2) {
	int n = str_1.length();
	int m = str_2.length();

	vector<int> current_row(n + 1);
	for (int i = 1; i <= n; i++) {
		current_row[i] = i;
	}

	for (int i = 1; i <= m; i++) {
		vector<int> previous_row = current_row;
		current_row[0] = i;
		for (int j = 1; j <= n; j++) {
			current_row[j] = min(previous_row[j] + 1, min(current_row[j - 1] + 1, 
				(str_1[j - 1] != str_2[i - 1]) ? previous_row[j - 1] + 1 : previous_row[j - 1]));
		}
	}
	return current_row[n];
}

int DamerauLevenshteinDistance(const string& source, const string& target) {

	int n = source.length();
	int m = target.length();
	vector<vector<int>> distanceMatrix(n + 1, vector<int>(m + 1));

	for (int i = 0; i <= n; ++i) {
		distanceMatrix[i][0] = i;
	}
	for (int j = 0; j <= m; ++j) {
		distanceMatrix[0][j] = j;
	}

	for (int i = 1; i <= n; ++i) {
		for (int j = 1; j <= m; ++j) {

			int substitutionCost = (source[i - 1] != target[j - 1]);

			int deletionCost = distanceMatrix[i - 1][j] + 1;
			int insertionCost = distanceMatrix[i][j - 1] + 1;
			int substitutionOrEqualityCost = distanceMatrix[i - 1][j - 1] + substitutionCost;

			//  транспозиция, если это возможно
			if (i > 1 && j > 1 && source[i - 1] == target[j - 2] && source[i - 2] == target[j - 1]) {
				distanceMatrix[i][j] = min({ deletionCost, insertionCost, substitutionOrEqualityCost, distanceMatrix[i - 2][j - 2] + substitutionCost });
			}
			else {
				distanceMatrix[i][j] = min({ deletionCost, insertionCost, substitutionOrEqualityCost });
			}
		}
	}

	for (int i = 0; i <= source.length(); ++i) {
		for (int j = 0; j <= target.length(); ++j) {
			cout << distanceMatrix[i][j] << " ";
		}
		cout << endl;
	}

	return distanceMatrix[n][m];
}

\begin{lstlisting}[caption={Расстояние Дамерау-Левенштейна рекурсивный метод, используя три строки матрицы},label={lst:label3}]
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <ctime>

using namespace std;

string RandomWord(int length) {
	static const char alphabet[] = "абвгдеёжзиклмнопрстуфхцчшщъыьэюя";
	string result;

	for (int i = 0; i < length; ++i) {
		int randomIndex = rand() % (sizeof(alphabet) - 1);
		result += alphabet[randomIndex];
	}

	return result;
}

int damerauLevenshteinDistance(const string& str1, const string& str2, int i, int j, vector<vector<int>>& memo) {
	if (i == 0)
		return j;

	if (j == 0)
		return i;

	if (memo[i][j] != -1)
		return memo[i][j];

	int cost = (str1[i - 1] == str2[j - 1]) ? 0 : 1;

	int deletee = damerauLevenshteinDistance(str1, str2, i - 1, j, memo) + 1;
	int insert = damerauLevenshteinDistance(str1, str2, i, j - 1, memo) + 1;
	int substitution = damerauLevenshteinDistance(str1, str2, i - 1, j - 1, memo) + cost;

	int distance = min({ deletee, insert, substitution });

	if (i > 1 && j > 1 && str1[i - 1] == str2[j - 2] && str1[i - 2] == str2[j - 1]) {
		int transposition = damerauLevenshteinDistance(str1, str2, i - 2, j - 2, memo) + cost;
		distance = min(distance, transposition);
	}

	memo[i][j] = distance;
	return distance;
}

int damerauLevenshteinDistance(const string& str1, const string& str2) {
	int m = str1.length();
	int n = str2.length();

	vector<vector<int>> memo(m + 1, vector<int>(n + 1, -1));

	return damerauLevenshteinDistance(str1, str2, m, n, memo);
}

			cout << "Расстояние Дамерау-Левенштейна: " << damerauLevenshteinDistance(str1, str2) << endl;

		}
		cout << "Среднее значение работы процессора: " << medtime / num;
	}

\justifying
Нетрудно заметить, что затраченное время возрастсает с увеличением количества вычислений. Поскольку при заполнении матрицы требуется сделать $3^{n \cdot n}$ вычислений, где n - количество символов в слове, то необходимо расходовать большое количество времени. Что касается алгоритма Дамерау-Левенштейна с рекурсией с кэшем, то затраченного времени и расходованной памяти, безусловно, становится гораздо меньше. Однако при больших объемах информации лидирует алгоритм Левенштейна без рекурсии с помощью двух строк. Данный алгоритм универсален, так как хранит в разы меньше информации по сравнению с другими алгоритмами и тратит на вычисления меньше времени. Он будет являться самым ``выигрышным''.
